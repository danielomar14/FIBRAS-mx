\section*{Objetivos del Proyecto}
\addcontentsline{toc}{section}{Objetivos del Proyecto}

\begin{enumerate}
  \item \textbf{Paso 0 -- Datos}: Construir un pipeline robusto de adquisici\'{o}n y validaci\'{o}n
        de datos hist\'{o}ricos (precios diarios + distribuciones) para las 15 FIBRAs activas,
        con cobertura objetivo $\geq 80\%$ por ticker.

  \item \textbf{Experimento 1}: Backtestear 11 estrategias de inversi\'{o}n (E0--E10) sobre el
        per\'{i}odo 2016--2025 con capital inicial de \$200{,}000 MXN, rebalanceo trimestral y
        reinversi\'{o}n DRIP al 100\%.

  \item \textbf{Experimento 2}: Optimizar la estrategia ganadora mediante un Algoritmo Gen\'{e}tico
        (cromosoma $\sim$30 genes, poblaci\'{o}n 500, 200 generaciones) con validaci\'{o}n
        anti-overfitting rigurosa (train/val/test, walk-forward, bootstrap).

  \item \textbf{App Streamlit}: Exponer resultados y exploraci\'{o}n interactiva en una aplicaci\'{o}n
        multipágina accesible localmente.
\end{enumerate}

\subsection{Objetivos de la sesi\'{o}n}
\begin{itemize}
  \item Crear entorno conda \texttt{fibras-mx} (Python 3.12).
  \item Definir estructura de directorios del repositorio.
  \item Crear \texttt{requirements.txt} con todas las dependencias.
  \item Iniciar la bit\'{a}cora del proyecto.
  \item Dise\~{n}ar arquitectura del pipeline de datos (Paso 0).
\end{itemize}

\subsection{Acciones realizadas}

\begin{enumerate}
  \item \textbf{Entorno conda} creado exitosamente:
\begin{lstlisting}[language=bash]
conda create -n fibras-mx python=3.12 -y
conda activate fibras-mx
pip install -r requirements.txt
\end{lstlisting}

  \item \textbf{Estructura de directorios} inicializada:
\begin{lstlisting}
FIBRAS-mx/
├── data/{raw/manual, processed, cache}
├── src/{data, backtest, genetic}
├── app/pages/
├── notebooks/
├── results/{charts, reports}
└── docs/
\end{lstlisting}

  \item \textbf{Fuentes de datos} priorizadas (estrategia en cascada):
  \begin{enumerate}[label=\arabic*.]
    \item \texttt{yfinance} -- precios diarios y distribuciones con sufijo \texttt{.MX}
    \item Scraping \texttt{fibrasmx.com} -- m\'{e}tricas trimestrales (ocupaci\'{o}n, LTV, payout)
    \item Banxico SIE / BMV -- gaps en FIBRAs peque\~{n}as
    \item CSVs manuales -- NEXT25 (solo desde 2025) y FIBRAs sin cobertura
  \end{enumerate}

  \item \textbf{Arquitectura Streamlit} dise\~{n}ada:
  \begin{itemize}
    \item P\'{a}gina 0: Dashboard calidad de datos (cobertura, huecos, fuente por ticker)
    \item P\'{a}gina 1: Backtest E0--E10, tabla comparativa, equity curves
    \item P\'{a}gina 2: Algoritmo Gen\'{e}tico, convergencia, cromosoma ganador
  \end{itemize}
\end{enumerate}

\subsection{Hallazgos}
\begin{itemize}
  \item \textbf{NEXT25} solo existe desde 2025 -- ajustar horizontes de backtest por ticker.
  \item \textbf{Terrafina (TERRA13)} fue absorbida por Fibra Prologis en 2024; no se incluye como ticker independiente.
  \item Yahoo Finance puede tener ajuste de dividendos inconsistente en FIBRAs: se usar\'{a} precio \textbf{sin ajustar} + dividendos por separado para calcular retorno total manualmente.
  \item FIBRAs de baja liquidez (FPLUS16, EDUCA18, STORAGE18, FSITES20) pueden tener huecos en la serie hist\'{o}rica.
\end{itemize}

\subsection{Pendientes (pr\'{o}xima sesi\'{o}n)}

\subsection{Acciones realizadas}
\begin{enumerate}
  \item \textbf{\texttt{src/backtest/metrics.py}} -- M\'{o}dulo de m\'{e}tricas de performance:
    \texttt{cagr()}, \texttt{annualized\_volatility()}, \texttt{max\_drawdown()},
    \texttt{sharpe()}, \texttt{calmar()}, \texttt{annual\_returns()},
    \texttt{consistency()}, \texttt{drawdown\_series()}, \texttt{summary()}.
    Rf diaria: $(1 + r/100)^{1/365} - 1$ sobre CETES 28d.

  \item \textbf{\texttt{src/backtest/engine.py}} -- Motor de simulaci\'{o}n diaria:
    \begin{itemize}
      \item Rebalanceo trimestral (primer d\'{i}a h\'{a}bil de ene/abr/jul/oct).
      \item DRIP 100\%: dividendos reinvertidos al precio de cierre del d\'{i}a ex-div.
      \item Costos: comisi\'{o}n 0.25\% $\times$ 1.16 IVA + slippage 0.10\% sobre monto operado.
      \item Capital inicial: \$200{,}000 MXN.
      \item Firma de estrategia: \texttt{fn(date, prices\_wide, dividends, metrics, banxico\_rates) -> dict[ticker, peso]}.
    \end{itemize}

  \item \textbf{\texttt{src/backtest/strategies.py}} -- 11 estrategias implementadas:
    \begin{center}
    \begin{tabular}{lll}
      \toprule
      C\'{o}d. & Nombre & Descripci\'{o}n \\
      \midrule
      E0  & Equiponderada          & Todas las FIBRAs disponibles, peso igual \\
      E1  & Grandes (proxy cap)    & Pondera por precio promedio 30d (proxy size) \\
      E2  & Momentum 12M           & Top 5 por retorno trailing 12 meses \\
      E3  & Momentum 3M            & Top 5 por retorno trailing 3 meses \\
      E4  & Baja volatilidad       & Top 5 con menor vol realizada 12M \\
      E5  & Alto yield             & Top 5 por dividendo TTM / precio \\
      E6  & Calidad (ocupaci\'{o}n) & Top 5 por ocupaci\'{o}n \'{u}ltima m\'{e}trica trimestral \\
      E7  & Valor (FFO yield)      & Top 5 por FFO/precio anualizado \\
      E8  & Momentum + Yield       & Top 5 por score compuesto (50/50) \\
      E9  & Contrarian             & Top 5 peores 12M (reversi\'{o}n a la media) \\
      E10 & Rotaci\'{o}n sectorial & Pondera industrial vs hotelero seg\'{u}n direcci\'{o}n de tasas \\
      \bottomrule
    \end{tabular}
    \end{center}

  \item \textbf{\texttt{app/pages/2\_estrategias.py}} -- P\'{a}gina Streamlit con 4 tabs:
    \begin{itemize}
      \item \textit{Curvas de valor}: equity curves + referencia CETES, m\'{e}tricas r\'{a}pidas.
      \item \textit{Drawdown}: serie de caída relativa con fill, tabla MaxDD.
      \item \textit{Retornos anuales}: heatmap + tabla coloreada por a\~{n}o y estrategia.
      \item \textit{Risk-return}: scatter CAGR vs Vol, coloreado por Sharpe.
    \end{itemize}
    Backtests cacheados con \texttt{@st.cache\_data(ttl=3600)} para carga instant\'{a}nea.
\end{enumerate}

\subsection{Hallazgos}
\begin{itemize}
  \item E9 (Contrarian) domina el per\'{i}odo: las FIBRAs que m\'{a}s cayeron tienden a recuperarse
        en el siguiente trimestre, especialmente post-COVID y post-pico de tasas.
  \item E3 (Momentum 3M) es destructivo de valor: el ruido a corto plazo en mercados
        ilíquidos genera rotaci\'{o}n excesiva y compra en picos.
  \item El Sharpe m\'{a}s alto es E4 (Baja vol, 0.288), mejor relaci\'{o}n riesgo-retorno ajustada.
  \item E7 (FFO yield) y E6 (Ocupaci\'{o}n) tienen MaxDD severo por concentraci\'{o}n en
        pocas FIBRAs; los datos trimestrales tienen gaps que limitan la selecci\'{o}n.
  \item Corregido bug de pandas 2.x: \texttt{fillna(method=''ffill'')} $\to$ \texttt{.ffill()}.
\end{itemize}
